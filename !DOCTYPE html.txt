    const BlurText = ({ text, className = "", delayBase = 0, once = true }) => {
        const [hasAnimated, setHasAnimated] = useState(false);
        const ref = useRef(null);
        useEffect(() => {
            const observer = new IntersectionObserver(([entry]) => {
                if (entry.isIntersecting && !hasAnimated) {
                    setHasAnimated(true);
                    observer.disconnect();
                }
            }, { threshold: 0.1 });
            if (ref.current) observer.observe(ref.current);
            return () => observer.disconnect();
        }, [hasAnimated]);
        const words = text.split(' ');
        return (
            <p ref={ref} className={`flex flex-wrap justify-center gap-x-[0.28em] ${className}`}>
                {words.map((word, i) => (
                    <motion.span
                        key={i}
                        initial={hasAnimated ? false : { filter: 'blur(10px)', opacity: 0, y: 50 }}
                        animate={hasAnimated ? { filter: 'blur(0px)', opacity: 1, y: 0 } : {}}
                        transition={{
                            duration: 0.7,
                            delay: delayBase + (i * 0.1),
                            ease: "easeOut",
                            times: [0, 0.5, 1],
                            keyframes: hasAnimated ? undefined : [
                                { filter: 'blur(10px)', opacity: 0, y: 50 },
                                { filter: 'blur(5px)', opacity: 0.5, y: -5 },
                                { filter: 'blur(0px)', opacity: 1, y: 0 }
                            ]
                        }}
                        style={{ display: 'inline-block', marginRight: '0.28em' }}
                    >
                        {word}
                    </motion.span>
                ))}
            </p>
        );
    };
    // ---------- FadingVideo 组件 (纯requestAnimationFrame 交叉淡变，手动循环) ----------
    const FadingVideo = ({ src, className, style = {}, scale120 = false, objectTop = false }) => {
        const videoRef = useRef(null);
        const rAFRef = useRef(null);
        const fadingOutRef = useRef(false);
        const currentOpacity = useRef(0);
        const fadeTo = useCallback((targetOpacity, durationMs, callback) => {
            if (rAFRef.current) cancelAnimationFrame(rAFRef.current);
            const startOpacity = currentOpacity.current;
            const delta = targetOpacity - startOpacity;
            const startTime = performance.now();
            const step = (now) => {
                const elapsed = now - startTime;
                let t = Math.min(1, elapsed / durationMs);
                let newOpacity = startOpacity + delta * t;
                if (videoRef.current) videoRef.current.style.opacity = newOpacity;
                currentOpacity.current = newOpacity;
                if (t < 1) rAFRef.current = requestAnimationFrame(step);
                else {
                    if (callback) callback();
                    rAFRef.current = null;
                }
            };
            rAFRef.current = requestAnimationFrame(step);
        }, []);
        useEffect(() => {
            const video = videoRef.current;
            if (!video) return;
            const onLoaded = () => {
                video.style.opacity = 0;
                currentOpacity.current = 0;
                video.play().catch(e=>console.log);
                fadeTo(1, 500);
            };
            const onTimeUpdate = () => {
                if (!fadingOutRef.current && video.duration - video.currentTime <= 0.55 && video.duration - video.currentTime > 0) {
                    fadingOutRef.current = true;
                    fadeTo(0, 500, () => {});
                }
            };
            const onEnded = () => {
                if (video) {
                    video.style.opacity = 0;
                    currentOpacity.current = 0;
                    setTimeout(() => {
                        video.currentTime = 0;
                        video.play().catch(e=>console.log);
                        fadingOutRef.current = false;
                        fadeTo(1, 500);
                    }, 100);
                }
            };
            video.addEventListener('loadeddata', onLoaded);
            video.addEventListener('timeupdate', onTimeUpdate);
            video.addEventListener('ended', onEnded);
            return () => {
                video.removeEventListener('loadeddata', onLoaded);
                video.removeEventListener('timeupdate', onTimeUpdate);
                video.removeEventListener('ended', onEnded);
                if (rAFRef.current) cancelAnimationFrame(rAFRef.current);
            };
        }, [fadeTo]);
        const combinedClasses = `absolute ${objectTop ? 'top-0 left-1/2 -translate-x-1/2 object-top' : 'inset-0 w-full h-full object-cover'} ${className}`;
        const videoStyle = scale120 ? { width: '120%', height: '120%', ...style } : style;
        return <video ref={videoRef} src={src} autoPlay muted playsInline preload="auto" loop={false} className={combinedClasses} style={{ opacity: 0, ...videoStyle }} />;

            </div>
        );
    };
